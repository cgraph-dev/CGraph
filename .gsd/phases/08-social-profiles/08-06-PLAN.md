1. Add `mutually_blocked?/2` to `apps/backend/lib/cgraph/accounts/friends/queries.ex`:
   ```elixir
   @doc "Check if either user has blocked the other (bidirectional)."
   @spec mutually_blocked?(String.t(), String.t()) :: boolean()
   def mutually_blocked?(user_a_id, user_b_id) do
     blocked?(user_a_id, user_b_id) || blocked?(user_b_id, user_a_id)
   end
   ```

2. Add a `get_blocked_user_ids/1` function to queries.ex:
   ```elixir
   @doc "Get all user IDs that are blocked by OR have blocked the given user."
   @spec get_blocked_user_ids(String.t()) :: [String.t()]
   def get_blocked_user_ids(user_id) do
     from(f in Friendship,
       where: (f.user_id == ^user_id or f.friend_id == ^user_id) and f.status == :blocked,
       select: fragment("CASE WHEN ? = ? THEN ? ELSE ? END", f.user_id, type(^user_id, Ecto.UUID), f.friend_id, f.user_id)
     )
     |> Repo.all()
   end
   ```

3. Enforce blocks in `conversation_channel.ex` — `handle_in("new_message")`:
   - Before processing the message, check if sender and recipient are mutually blocked:
     ```elixir
     # At the top of handle_in("new_message", ...)
     sender_id = socket.assigns.current_user.id
     # Get other participant(s) from conversation
     # If any participant is mutually blocked with sender, reject with error

       {:reply, {:error, %{reason: "blocked"}}, socket}
     else
       # ... existing message handling
     end
     ```
   - The conversation likely has participant IDs in socket assigns or can be fetched from conversation record. Check the `join/3` function to see how conversation membership is validated.
