Add or enhance these animation variants (append, don't replace existing):

```ts
import { springs, stagger } from '@cgraph/animation-constants';

/** Message entrance variant — direction-aware slide + fade */
export const messageEntranceVariants = {
  /** Sent message (slides from right) */
  sent: {
    initial: { opacity: 0, x: 20, scale: 0.97 },
    animate: { opacity: 1, x: 0, scale: 1 },
    transition: {
      type: 'spring' as const,
      stiffness: springs.snappy.stiffness,
      damping: springs.snappy.damping,
    },
  },
  /** Received message (slides from left) */
  received: {
    initial: { opacity: 0, x: -20, scale: 0.97 },
    animate: { opacity: 1, x: 0, scale: 1 },
    transition: {
      type: 'spring' as const,
      stiffness: springs.snappy.stiffness,
      damping: springs.snappy.damping,
    },
  },
} as const;

/** Stagger container for batch message arrivals */
export const messageListStagger = {
  animate: {
    transition: {
      staggerChildren: stagger.fast.staggerChildren,
    },
  },
} as const;

/** Typing indicator pulse */
export const typingIndicatorVariants = {
  animate: {
    opacity: [0.4, 1, 0.4],
    transition: {
      duration: 1.2,
      repeat: Infinity,
      ease: 'easeInOut',
    },
  },
} as const;
```

2. Wrap the message bubble with motion.div:

```tsx
const reducedMotion = useReducedMotion();

const variant = messageEntranceVariants[direction];

<motion.div
  initial={reducedMotion ? undefined : variant.initial}
  animate={variant.animate}
  transition={reducedMotion ? { duration: 0 } : variant.transition}
>
  {/* existing message bubble content */}
</motion.div>;
```
