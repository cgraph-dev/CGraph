## 🚀 Migration Guide

### For Developers

#### 1. Import New Components

**Before:**
```typescript
// No rich media support
import { VoiceMessagePlayer } from '@/components';
```

**After:**
```typescript
// Full rich media support
import { 
  VoiceMessagePlayer, 
  RichMediaEmbed 
} from '@/components';
```

#### 2. Use Rich Media in Messages

**Add to ConversationScreen.tsx:**
```typescript
<RichMediaEmbed 
  content={message.content}
  isOwnMessage={message.sender.id === user?.id}
  onLoad={() => scrollToBottom()}
/>
```

#### 3. Navigate to Premium Screens

**Add to navigation config:**
```typescript
<Stack.Screen 
  name="Premium" 
  component={PremiumScreen}
  options={{ title: 'CGraph Premium' }}
/>

<Stack.Screen 
  name="CoinShop" 
  component={CoinShopScreen}
  options={{ title: 'Coin Shop' }}
/>
```

#### 4. Integrate Payment Provider

**Stripe Example:**
```typescript
import { useStripe } from '@stripe/stripe-react-native';

const handleSubscribe = async (tier: PremiumTier) => {
  const { error, paymentIntent } = await stripe.confirmPayment({
    amount: tier.price[billingCycle] * 100,
    currency: 'usd',
  });
  
  if (!error) {
    // Update user subscription status
    await api.post('/subscriptions', { tier: tier.id });
  }
};
```

**RevenueCat Example (Recommended for Mobile):**
```typescript
import Purchases from 'react-native-purchases';

const handleSubscribe = async (tier: PremiumTier) => {
  try {
    const offerings = await Purchases.getOfferings();
    const packageId = tier.id;
    
    const { customerInfo } = await Purchases.purchasePackage(
      offerings.current[packageId]
    );
    
    // Check active entitlements
    if (customerInfo.entitlements.active[tier.id]) {
      // Subscription active
    }
  } catch (error) {
    // Handle error
  }
};
```
