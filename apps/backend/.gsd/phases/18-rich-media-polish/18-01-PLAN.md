### Task 1 — File Encryption Crypto Layer (E2EE-05 Foundation)

**What:** Add file-level AES-256-GCM encryption to `packages/crypto/` — a new `file-encryption.ts`
module that encrypts/decrypts arbitrary `ArrayBuffer` data. The existing `aes.ts` only handles
string data (`encryptAES`/`decryptAES` at L44/L79); files need binary support.

**Where:** `packages/crypto/src/file-encryption.ts`, `packages/crypto/src/index.ts`

**How:**

Create `packages/crypto/src/file-encryption.ts`:

```typescript
import { generateAESKey, importAESKey } from './aes';

export interface EncryptedFile {
  ciphertext: ArrayBuffer;
  iv: Uint8Array;
  fileKey: ArrayBuffer; // raw AES key bytes — caller wraps with ratchet key
}

export interface DecryptedFile {
  plaintext: ArrayBuffer;
}

/**
 * Encrypt a file with a random AES-256-GCM key.
 * Returns ciphertext + IV + raw file key (caller must wrap key with session ratchet key).
 */
export async function encryptFile(data: ArrayBuffer): Promise<EncryptedFile> {
  const fileKey = await generateAESKey();
  const iv = crypto.getRandomValues(new Uint8Array(12));
  const rawKey = await crypto.subtle.exportKey('raw', fileKey);

  const ciphertext = await crypto.subtle.encrypt({ name: 'AES-GCM', iv }, fileKey, data);

  return { ciphertext, iv, fileKey: rawKey };
}

/**
 * Decrypt a file given the raw AES key and IV.
 */
export async function decryptFile(
  ciphertext: ArrayBuffer,
  rawKey: ArrayBuffer,
  iv: Uint8Array
): Promise<DecryptedFile> {
  const key = await importAESKey(rawKey);
  const plaintext = await crypto.subtle.decrypt({ name: 'AES-GCM', iv }, key, ciphertext);
  return { plaintext };
}
```

Re-export from `packages/crypto/src/index.ts`:

```typescript

export type { EncryptedFile, DecryptedFile } from './file-encryption';
```

```typescript
import { encryptFile } from '@cgraph/crypto';
import { getSessionKey } from '../e2ee/session-manager';
import { encryptAES } from '@cgraph/crypto';

export async function encryptFileForUpload(
  file: File,
  conversationId: string
): Promise<{
  encryptedBlob: Blob;
  metadata: { encrypted_key: string; encryption_iv: string; sender_device_id: string };
}> {
  const buffer = await file.arrayBuffer();
  const { ciphertext, iv, fileKey } = await encryptFile(buffer);

  // Wrap file key with session ratchet key
  const sessionKey = await getSessionKey(conversationId);
  const wrappedKey = await encryptAES(
    new TextDecoder().decode(new Uint8Array(fileKey)),
    sessionKey
  );

  return {
    encryptedBlob: new Blob([ciphertext], { type: 'application/octet-stream' }),
    metadata: {
      encrypted_key: btoa(String.fromCharCode(...new Uint8Array(wrappedKey))),
      encryption_iv: btoa(String.fromCharCode(...iv)),
      sender_device_id: getDeviceId(),
    },
  };
}
```

```typescript
import { encryptAES, generateAESKey } from '@cgraph/crypto';

export async function encryptVoiceMetadata(
  waveform: number[],
  duration: number,
  sessionKey: CryptoKey
): Promise<{
  encrypted_waveform: string;
  encrypted_duration: string;
  waveform_iv: string;
  duration_iv: string;
  metadata_encrypted_key: string;
}> {
  const metadataKey = await generateAESKey();
  const waveformData = JSON.stringify(waveform);
  const durationData = String(duration);

  const encWaveform = await encryptAES(waveformData, metadataKey);
  const encDuration = await encryptAES(durationData, metadataKey);

  // Wrap metadata key with session ratchet key
  const rawKey = await crypto.subtle.exportKey('raw', metadataKey);
  const wrappedKey = await encryptAES(
    btoa(String.fromCharCode(...new Uint8Array(rawKey))),
    sessionKey
  );

  return {
    encrypted_waveform: encWaveform.ciphertext,
    encrypted_duration: encDuration.ciphertext,
    waveform_iv: encWaveform.iv,
    duration_iv: encDuration.iv,
    metadata_encrypted_key: wrappedKey.ciphertext,
  };
}
```

```elixir
defmodule CGraph.Messaging.ScheduledMessages do
  import Ecto.Query
  alias CGraph.Repo
  alias CGraph.Messaging.Message

  def list_scheduled(user_id, opts \\ []) do
    conversation_id = Keyword.get(opts, :conversation_id)

    Message
    |> where([m], m.sender_id == ^user_id and m.schedule_status == "scheduled")
    |> then(fn q ->
      if conversation_id, do: where(q, [m], m.conversation_id == ^conversation_id), else: q
    end)
    |> order_by([m], asc: m.scheduled_at)
    |> Repo.all()
  end

  def create_scheduled(user_id, attrs) do
    attrs
    |> Map.put("sender_id", user_id)
    |> Map.put("schedule_status", "scheduled")
    |> validate_scheduled_at()
    |> case do
      {:ok, attrs} -> Messaging.create_message(attrs)
      {:error, reason} -> {:error, reason}
    end
  end

  def update_scheduled(user_id, message_id, attrs) do
    with {:ok, message} <- get_user_scheduled(user_id, message_id),
         {:ok, attrs} <- validate_scheduled_at(attrs) do
      message |> Message.changeset(attrs) |> Repo.update()
    end
  end

  def cancel_scheduled(user_id, message_id) do
    with {:ok, message} <- get_user_scheduled(user_id, message_id) do
      message
      |> Ecto.Changeset.change(schedule_status: "cancelled")
      |> Repo.update()
    end
  end

  defp validate_scheduled_at(%{"scheduled_at" => at} = attrs) when not is_nil(at) do
    case DateTime.compare(at, DateTime.utc_now()) do
      :gt -> {:ok, attrs}
      _ -> {:error, :scheduled_at_must_be_future}
    end
  end

  defp get_user_scheduled(user_id, message_id) do
    Message
    |> where([m], m.id == ^message_id and m.sender_id == ^user_id and m.schedule_status == "scheduled")
    |> Repo.one()
    |> case do
      nil -> {:error, :not_found}
      message -> {:ok, message}
    end
  end
end
```

```elixir
defmodule CGraphWeb.API.V1.ScheduledMessageController do
  use CGraphWeb, :controller

  alias CGraph.Messaging.ScheduledMessages

  def index(conn, params) do
    user = conn.assigns.current_user
    opts = if params["conversation_id"], do: [conversation_id: params["conversation_id"]], else: []
    messages = ScheduledMessages.list_scheduled(user.id, opts)
    render(conn, :index, messages: messages)
  end

  def create(conn, %{"message" => params}) do
    user = conn.assigns.current_user
    case ScheduledMessages.create_scheduled(user.id, params) do
      {:ok, message} -> conn |> put_status(:created) |> render(:show, message: message)
      {:error, reason} -> conn |> put_status(:unprocessable_entity) |> json(%{error: reason})
    end
  end

  def update(conn, %{"id" => id, "message" => params}) do
    user = conn.assigns.current_user
    case ScheduledMessages.update_scheduled(user.id, id, params) do
      {:ok, message} -> render(conn, :show, message: message)
      {:error, :not_found} -> conn |> put_status(:not_found) |> json(%{error: "not_found"})
      {:error, reason} -> conn |> put_status(:unprocessable_entity) |> json(%{error: reason})
    end
  end

  def delete(conn, %{"id" => id}) do
    user = conn.assigns.current_user
    case ScheduledMessages.cancel_scheduled(user.id, id) do
      {:ok, _} -> send_resp(conn, :no_content, "")
      {:error, :not_found} -> conn |> put_status(:not_found) |> json(%{error: "not_found"})
    end
  end
end
```
