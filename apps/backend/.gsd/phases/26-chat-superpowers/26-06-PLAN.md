**Create `lib/cgraph/messaging/emoji.ex`:**

```elixir
defmodule CGraph.Messaging.Emoji do
  @moduledoc "Full Unicode 16.0 emoji catalog, search, and validation."

  @emoji_data File.read!("priv/data/emoji_unicode16.json") |> Jason.decode!()
  @emoji_set @emoji_data |> Enum.map(& &1["emoji"]) |> MapSet.new()
  @categories @emoji_data |> Enum.group_by(& &1["category"])

  @skin_tones %{
    light: "\u{1F3FB}",
    medium_light: "\u{1F3FC}",
    medium: "\u{1F3FD}",
    medium_dark: "\u{1F3FE}",
    dark: "\u{1F3FF}"
  }

  def valid_emoji?(emoji) do
    # Strip skin tone modifiers before checking
    base = strip_skin_tone(emoji)
    MapSet.member?(@emoji_set, base) or MapSet.member?(@emoji_set, emoji)
  end

  def search(query, opts \\ []) do
    limit = Keyword.get(opts, :limit, 50)
    category = Keyword.get(opts, :category)
    query_lower = String.downcase(query)

    @emoji_data
    |> maybe_filter_category(category)
    |> Enum.filter(fn e ->
      String.contains?(e["name"], query_lower) or
      Enum.any?(e["keywords"] || [], &String.contains?(&1, query_lower))
    end)
    |> Enum.take(limit)
  end

  def list_by_category(category), do: Map.get(@categories, category, [])
  def categories, do: Map.keys(@categories)
  def all, do: @emoji_data
  def skin_tone_variants(emoji), do: ... # return variants if skin_tone_support: true

  defp strip_skin_tone(emoji), do: String.replace(emoji, ~r/[\x{1F3FB}-\x{1F3FF}]/, "")
  defp maybe_filter_category(data, nil), do: data
  defp maybe_filter_category(data, cat), do: Enum.filter(data, &(&1["category"] == cat))
end
```
