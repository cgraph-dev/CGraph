<task id="3" title="Build coin checkout module">
Create `apps/backend/lib/cgraph/shop/coin_checkout.ex`:

```elixir
defmodule CGraph.Shop.CoinCheckout do
  alias CGraph.Shop.{CoinBundles, CoinPurchase}
  alias CGraph.Repo

  def create_checkout_session(user, bundle_id) do
    bundle = CoinBundles.get_bundle(bundle_id)

    # Create pending purchase record
    {:ok, purchase} = Repo.insert(%CoinPurchase{
      user_id: user.id,
      bundle_id: bundle_id,
      coins_awarded: bundle.coins,
      price_cents: bundle.price_cents,
      status: "pending"
    })

    # Create Stripe Checkout Session (mode: payment, NOT subscription)
    params = %{
      mode: "payment",
      customer: user.stripe_customer_id,
      line_items: [%{price: bundle.stripe_price_id, quantity: 1}],
      metadata: %{user_id: user.id, bundle_id: bundle_id, purchase_id: purchase.id},
      success_url: success_url(purchase.id),
      cancel_url: cancel_url()
    }

    case Stripe.Checkout.Session.create(params) do
      {:ok, session} ->
        Repo.update(Ecto.Changeset.change(purchase, stripe_session_id: session.id))
        {:ok, %{checkout_url: session.url, session_id: session.id}}
      {:error, error} ->
        Repo.update(Ecto.Changeset.change(purchase, status: "failed"))
        {:error, error}
    end
  end

  def fulfill_purchase(session_id) do
    # IDEMPOTENCY: Check if already fulfilled
    case Repo.get_by(CoinPurchase, stripe_session_id: session_id) do
      %CoinPurchase{status: "completed"} ->
        {:ok, :already_fulfilled}
      %CoinPurchase{} = purchase ->
        # Award coins via Gamification context
        # IMPORTANT: award_coins/4 requires %User{} struct (not UUID) and String.t() type (not atom)
        user = CGraph.Repo.get!(CGraph.Accounts.User, purchase.user_id)
        {:ok, _} = CGraph.Gamification.award_coins(user, purchase.coins_awarded, "purchase")
        Repo.update(Ecto.Changeset.change(purchase, status: "completed", fulfilled_at: DateTime.utc_now()))
      nil ->
        {:error, :purchase_not_found}
    end
  end
end
```

Wire `fulfill_purchase/1` into StripeWebhookController's `checkout.session.completed` handler —
