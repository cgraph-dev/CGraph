export function LottieRenderer({
  codepoint,
  emoji,
  size = 32,
  autoplay = false,
  loop = false,
  fallbackSrc,
}: LottieRendererProps) {
  const { source, isLoading, animationRef } = useLottie({ codepoint, autoplay, loop });

  if (reducedMotion || !source) {
    return <Image source={{ uri: fallbackSrc }} style={{ width: size, height: size }} />;
  }

  return (
    <LottieView
      ref={animationRef}
      source={typeof source === 'string' ? { uri: source } : source}
      style={{ width: size, height: size }}
      autoPlay={autoplay}
      loop={loop}
      renderMode="AUTOMATIC" // Uses hardware rendering when available
    />
  );
}
