<task id="2" title="Build IAP validator module">
Create `apps/backend/lib/cgraph/subscriptions/iap_validator.ex`:

```elixir
defmodule CGraph.Subscriptions.IAPValidator do
  @moduledoc """
  Validates IAP receipts from Apple App Store and Google Play Store.
  Dispatches to platform-specific validation logic.
  """

  alias CGraph.Subscriptions.ReceiptValidation
  alias CGraph.Repo

  @doc "Validate a receipt and activate subscription if valid."
  def validate_receipt(user, %{"platform" => "apple"} = params) do
    validate_apple(user, params)
  end

  def validate_receipt(user, %{"platform" => "google"} = params) do
    validate_google(user, params)
  end

  # Apple: Use App Store Server API v2
  # POST https://api.storekit.itunes.apple.com/inApps/v1/transactions/{transactionId}
  # Verify JWS signed transaction
  defp validate_apple(user, %{"transaction_id" => txn_id, "receipt_data" => receipt}) do
    # 1. Check idempotency — already validated this transaction?
    case Repo.get_by(ReceiptValidation, platform: "apple", original_transaction_id: txn_id) do
      %ReceiptValidation{validation_status: "valid"} = existing ->
        {:ok, existing}  # Already validated — return success
      _ ->
        # 2. Call Apple App Store Server API
        case call_apple_api(txn_id) do
          {:ok, transaction_info} ->
            # 3. Store receipt and activate subscription
            store_and_activate(user, "apple", txn_id, receipt, transaction_info)
          {:error, reason} ->
            {:error, reason}
        end
    end
  end

  # Google: Use Play Developer API v3
  # GET https://androidpublisher.googleapis.com/androidpublisher/v3/applications/{package}/purchases/subscriptions/{subscriptionId}/tokens/{token}
  defp validate_google(user, %{"purchase_token" => token, "product_id" => product_id}) do
    case Repo.get_by(ReceiptValidation, platform: "google", original_transaction_id: token) do
      %ReceiptValidation{validation_status: "valid"} = existing ->
        {:ok, existing}
      _ ->
        case call_google_api(product_id, token) do
          {:ok, subscription_info} ->
            store_and_activate(user, "google", token, token, subscription_info)
          {:error, reason} ->
            {:error, reason}
        end
    end
  end

  defp store_and_activate(user, platform, txn_id, receipt_data, info) do
    # Insert receipt record
    {:ok, receipt} = Repo.insert(%ReceiptValidation{
      user_id: user.id,
      platform: platform,
      product_id: info.product_id,
      original_transaction_id: txn_id,
      receipt_data: receipt_data,
      validation_status: "valid",
      expires_at: info.expires_at,
      purchase_date: info.purchase_date,
      environment: info.environment,
      auto_renewing: info.auto_renewing
    })

    # Activate subscription on user
    # IMPORTANT: activate_subscription/2 expects these exact keys:
    #   params.tier (string), params.current_period_end (unix int),
    #   params.stripe_subscription_id, params.stripe_customer_id
    tier = product_id_to_tier(info.product_id)
    {:ok, _user} = CGraph.Subscriptions.activate_subscription(user, %{
      tier: tier,
      current_period_end: DateTime.to_unix(info.expires_at),
      stripe_subscription_id: nil,
      stripe_customer_id: user.stripe_customer_id
    })

    # Also set IAP-specific fields (cast by extended subscription_changeset from 17-01 task 0)
    {:ok, _user} = user
    |> User.subscription_changeset(%{iap_provider: platform, iap_transaction_id: txn_id})
    |> Repo.update()

    {:ok, receipt}
  end
end
```

Include `call_apple_api/1` and `call_google_api/2` using `CGraph.HTTP.Services.Stripe` pattern

<task id="3" title="Build IAP controller">
Create `apps/backend/lib/cgraph_web/controllers/iap_controller.ex`:

Endpoints:

1. `POST /api/v1/iap/validate` — Receive receipt from mobile, validate, activate
2. `POST /api/v1/iap/restore` — Restore purchases (re-validate all stored receipts for user)
3. `POST /api/v1/iap/notifications/apple` — Apple Server-to-Server Notification v2
4. `POST /api/v1/iap/notifications/google` — Google Real-Time Developer Notification (RTDN)

Apple S2S notifications handle: `DID_RENEW`, `DID_CHANGE_RENEWAL_STATUS`, `REFUND`, `EXPIRED`,
`REVOKE`. Google RTDN handles: `SUBSCRIPTION_RENEWED`, `SUBSCRIPTION_CANCELED`,
`SUBSCRIPTION_EXPIRED`, `SUBSCRIPTION_PAUSED`.

Both notification endpoints must verify authenticity (Apple: signed JWS, Google: OAuth token) before
processing. </task>

```typescript
import {
  initConnection,
  endConnection,
  getSubscriptions,
  requestSubscription,
  finishTransaction,
  purchaseUpdatedListener,
  purchaseErrorListener,
  type ProductPurchase,
  type Subscription,
} from 'react-native-iap';
import { Platform } from 'react-native';
import { api } from '../../lib/api';

const SUBSCRIPTION_SKUS =
  Platform.select({
    ios: ['com.cgraph.premium.monthly', 'com.cgraph.premium.yearly'],
    android: ['com.cgraph.premium.monthly', 'com.cgraph.premium.yearly'],
  }) ?? [];

class IAPService {
  private products: Subscription[] = [];
  private purchaseListener: ReturnType<typeof purchaseUpdatedListener> | null = null;
  private errorListener: ReturnType<typeof purchaseErrorListener> | null = null;

  async initialize(): Promise<void> {
    await initConnection();
    // Set up purchase listeners
    this.purchaseListener = purchaseUpdatedListener(this.handlePurchase.bind(this));
    this.errorListener = purchaseErrorListener((error) => {
      console.warn('IAP purchase error:', error);
    });
  }

  async loadProducts(): Promise<Subscription[]> {
    this.products = await getSubscriptions({ skus: SUBSCRIPTION_SKUS });
    return this.products;
  }

  async purchaseSubscription(productId: string): Promise<void> {
    await requestSubscription({ sku: productId });
    // Purchase listener handles the result via handlePurchase
  }

  private async handlePurchase(purchase: ProductPurchase): Promise<void> {
    if (purchase.transactionId) {
      // Send receipt to backend for validation
      await this.validateReceipt(purchase);
      // Finish transaction (REQUIRED by both Apple and Google)
      await finishTransaction({ purchase, isConsumable: false });
    }
  }

  private async validateReceipt(purchase: ProductPurchase): Promise<void> {
    const platform = Platform.OS === 'ios' ? 'apple' : 'google';
    await api.post('/api/v1/iap/validate', {
      platform,
      transaction_id: purchase.transactionId,
      receipt_data: purchase.transactionReceipt,
      product_id: purchase.productId,
      purchase_token: purchase.purchaseToken, // Google-specific
    });
  }

  async restorePurchases(): Promise<void> {
    await api.post('/api/v1/iap/restore');
  }

  async destroy(): Promise<void> {
    this.purchaseListener?.remove();
    this.errorListener?.remove();
    await endConnection();
  }
}

export const iapService = new IAPService();
```
