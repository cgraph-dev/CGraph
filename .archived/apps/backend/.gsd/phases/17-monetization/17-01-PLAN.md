<task id="2" title="Build idempotency module">
Create `apps/backend/lib/cgraph/subscriptions/idempotency.ex`:

```elixir
defmodule CGraph.Subscriptions.Idempotency do
  @moduledoc "Idempotent webhook event processing."

  alias CGraph.Subscriptions.WebhookEvent
  alias CGraph.Repo

  @doc """
  Process a Stripe event exactly once. Returns {:ok, result} if processed,
  {:already_processed, event} if duplicate, or {:error, reason} on failure.
  """
  def process_once(%Stripe.Event{id: event_id, type: event_type} = event, handler_fn) do
    case Repo.insert(
      WebhookEvent.changeset(%WebhookEvent{}, %{
        stripe_event_id: event_id,
        event_type: event_type,
        payload: serialize_event(event),
        processed_at: DateTime.utc_now()
      }),
      on_conflict: :nothing,
      conflict_target: :stripe_event_id
    ) do
      {:ok, %{id: nil}} ->
        # on_conflict: :nothing returns nil id when conflict
        {:already_processed, event_id}
      {:ok, webhook_event} ->
        # New event — process it
        try do
          result = handler_fn.(event)
          {:ok, result}
        rescue
          e ->
            Repo.update(WebhookEvent.changeset(webhook_event, %{
              status: "failed",
              error_message: Exception.message(e)
            }))
            {:error, e}
        end
      {:error, changeset} ->
        {:error, changeset}
    end
  end
end
```

Key design: Uses Postgres unique constraint + `on_conflict: :nothing` for lock-free idempotency. No
race conditions. </task>

  {:ok, _result} ->

  {:already_processed, event_id} ->
    Logger.info("webhook_duplicate_skipped", event_id: event_id)

Pattern follows LevelGatePlug from Phase 16:

```elixir
defmodule CGraphWeb.Plugs.PremiumGatePlug do
  @moduledoc "Gate endpoints by subscription tier."
  import Plug.Conn

  def init(opts), do: opts

  def call(conn, min_tier: min_tier) do
    if Application.get_env(:cgraph, :bypass_premium_gates, false) do
      conn
    else
      do_gate(conn, min_tier)
    end
  end

  defp do_gate(conn, min_tier) do
    user = conn.assigns[:current_user]
    cond do
      is_nil(user) -> conn
      tier_sufficient?(user.subscription_tier, min_tier) -> conn
      true ->
        conn
        |> put_resp_content_type("application/json")
        |> send_resp(403, Jason.encode!(%{
          error: "tier_required",
          message: "Upgrade to #{min_tier} to access this feature",
          required_tier: min_tier,
          current_tier: user.subscription_tier
        }))
        |> halt()
    end
  end

  defp tier_sufficient?(current, required) do
    tier_level(current) >= tier_level(required)
  end

  defp tier_level("enterprise"), do: 3
  defp tier_level("premium"), do: 2
  defp tier_level("free"), do: 1
  defp tier_level(_), do: 0
end
```
