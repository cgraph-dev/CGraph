defmodule CGraph.Creators.ConnectOnboarding do
  @moduledoc "Manages Stripe Connect Express account onboarding for forum creators."

  def create_connect_account(user) do
    # Create Stripe Connect Express account
    params = %{
      type: "express",
      email: user.email,
      metadata: %{cgraph_user_id: user.id},
      capabilities: %{
        card_payments: %{requested: true},
        transfers: %{requested: true}
      }
    }

    case Stripe.Account.create(params) do
      {:ok, account} ->
        # Store connect_id on user — use creator_changeset (added by 17-01 task 0)
        {:ok, _} = user
          |> CGraph.Accounts.User.creator_changeset(%{
            stripe_connect_id: account.id,
            creator_status: "pending"
          })
          |> CGraph.Repo.update()
        # Generate onboarding link
        create_account_link(account.id)
      {:error, error} ->
        {:error, error}
    end
  end

  def create_account_link(connect_account_id) do
    Stripe.AccountLink.create(%{
      account: connect_account_id,
      refresh_url: "#{base_url()}/creator/onboard/refresh",
      return_url: "#{base_url()}/creator/onboard/complete",
      type: "account_onboarding"
    })
  end

  def check_account_status(connect_account_id) do
    case Stripe.Account.retrieve(connect_account_id) do
      {:ok, account} ->
        %{
          charges_enabled: account.charges_enabled,
          payouts_enabled: account.payouts_enabled,
          details_submitted: account.details_submitted,
          requirements: account.requirements
        }
      {:error, e} -> {:error, e}
    end
  end
end

defmodule CGraph.Creators.PaidSubscription do
  @platform_fee_percent 15  # CGraph takes 15%

  def subscribe_to_paid_forum(subscriber, forum) do
    creator = Repo.get!(User, forum.owner_id)

    # Ensure subscriber has Stripe customer
    subscriber = ensure_stripe_customer(subscriber)

    # Create subscription with application fee
    params = %{
      customer: subscriber.stripe_customer_id,
      items: [%{price_data: %{
        currency: forum.subscription_currency,
        unit_amount: forum.subscription_price_cents,
        recurring: %{interval: "month"},
        product_data: %{name: "#{forum.name} - Premium Access"}
      }}],
      application_fee_percent: @platform_fee_percent,
      transfer_data: %{destination: creator.stripe_connect_id},
      metadata: %{
        forum_id: forum.id,
        subscriber_id: subscriber.id,
        creator_id: creator.id,
        type: "paid_forum"
      }
    }

    case Stripe.Subscription.create(params) do
      {:ok, stripe_sub} ->
        # Create local record
        Repo.insert(%PaidForumSubscription{
          forum_id: forum.id,
          subscriber_id: subscriber.id,
          creator_id: creator.id,
          stripe_subscription_id: stripe_sub.id,
          price_cents: forum.subscription_price_cents,
          status: "active",
          current_period_start: DateTime.from_unix!(stripe_sub.current_period_start),
          current_period_end: DateTime.from_unix!(stripe_sub.current_period_end)
        })
      {:error, e} -> {:error, e}
    end
  end

  def cancel_paid_subscription(subscription) do
    Stripe.Subscription.update(subscription.stripe_subscription_id, %{cancel_at_period_end: true})
    Repo.update(Ecto.Changeset.change(subscription, status: "canceled", canceled_at: DateTime.utc_now()))
  end

  def has_active_subscription?(user_id, forum_id) do
    Repo.exists?(
      from s in PaidForumSubscription,
      where: s.subscriber_id == ^user_id and s.forum_id == ^forum_id
        and s.status in ["active", "canceled"]
        and s.current_period_end > ^DateTime.utc_now()
    )
  end
end

defmodule CGraph.Creators.Earnings do
  def record_earning(creator_id, %{} = params) do
    gross = params.amount_cents
    platform_fee = div(gross * platform_fee_percent(), 100)
    net = gross - platform_fee

    Repo.insert(%CreatorEarning{
      creator_id: creator_id,
      forum_id: params.forum_id,
      subscriber_id: params.subscriber_id,
      gross_amount_cents: gross,
      platform_fee_cents: platform_fee,
      net_amount_cents: net,
      stripe_payment_intent_id: params.payment_intent_id,
      period_start: params.period_start,
      period_end: params.period_end
    })
  end

  def get_balance(creator_id) do
    total_earned = sum earnings where creator_id (net_amount_cents)
    total_paid_out = sum payouts where creator_id and status = "completed" (amount_cents)
    %{
      total_earned_cents: total_earned,
      total_paid_out_cents: total_paid_out,
      available_balance_cents: total_earned - total_paid_out
    }
  end

  def get_stats(creator_id, period \\ :last_30_days) do
    %{
      subscriber_count: count active paid_forum_subscriptions for creator,
      mrr_cents: sum of active subscription prices,
      churn_rate: canceled / (canceled + active) in period,
      earnings_over_time: grouped by month,
      top_forums: by subscriber count
    }
  end
end

defmodule CGraph.Creators.Payout do
  @minimum_payout_cents 1000  # $10 minimum

  def request_payout(creator) do
    balance = Earnings.get_balance(creator.id)

    cond do
      balance.available_balance_cents < @minimum_payout_cents ->
        {:error, :below_minimum}
      creator.creator_status != "active" ->
        {:error, :account_not_active}
      true ->
        create_transfer(creator, balance.available_balance_cents)
    end
  end

  defp create_transfer(creator, amount_cents) do
    # Create Stripe Transfer to connected account
    case Stripe.Transfer.create(%{
      amount: amount_cents,
      currency: "usd",
      destination: creator.stripe_connect_id,
      metadata: %{creator_id: creator.id}
    }) do
      {:ok, transfer} ->
        Repo.insert(%CreatorPayout{
          creator_id: creator.id,
          amount_cents: amount_cents,
          stripe_transfer_id: transfer.id,
          status: "processing"
        })
      {:error, e} ->
        {:error, e}
    end
  end
end
