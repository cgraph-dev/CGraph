**Create `lottie-renderer.tsx`:**

```tsx
export function LottieRenderer({
  codepoint,
  emoji,
  size = 32,
  autoplay = false,
  loop = false,
  playOnHover = true,
  className,
  fallbackSrc, // WebP URL for fallback
}: LottieRendererProps) {
  const containerRef = useRef<HTMLDivElement>(null);
  const { isLoaded, error } = useLottie({ codepoint, containerRef, autoplay, loop, playOnHover });

  // If reduced motion or Lottie load fails, render static WebP
  if (prefersReducedMotion || error) {
    return <img src={fallbackSrc} alt={emoji} width={size} height={size} />;
  }

  return (
    <div
      ref={containerRef}
      className={className}
      style={{ width: size, height: size }}
      aria-label={emoji}
      role="img"
    />
  );
}
```
