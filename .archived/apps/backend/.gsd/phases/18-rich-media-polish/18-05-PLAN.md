## Goal

Harden the moderation system: wire the AI auto-action pipeline for high-confidence detections,
extend the admin dashboard with rich metrics (trends, leaderboards, resolution rates), verify the
appeal system end-to-end, and add bulk moderation actions. Most backend infrastructure exists
(moderation.ex 87L facade, enforcement.ex 312L, appeals.ex 127L, stats.ex 61L, ai/moderation.ex
135L); this plan completes the automation, metrics, and polish layers.

Tasks proceed: (0) moderation audit log schema, (1) AI auto-action pipeline, (2) extended moderation
stats, (3) admin dashboard metrics UI, (4) bulk moderation actions, (5) appeal system verification +
email notifications, (6) integration test.

Create `auto_action.ex`:

```elixir
defmodule CGraph.Moderation.AutoAction do
  alias CGraph.Moderation.{Enforcement, Reports, AuditLogs}
  alias CGraph.AI.Moderation, as: AIModeration
  alias CGraph.Repo

  @high_confidence_threshold 0.9
  @auto_action_categories ~w(spam hate_speech violence sexual)a

  @doc """
  Process a content item through AI moderation and auto-action if high confidence.
  Called asynchronously via ModerationWorker.
  """
  def process(content, opts \\ []) do
    user_id = Keyword.fetch!(opts, :user_id)
    target_type = Keyword.get(opts, :type, "message")
    target_id = Keyword.fetch!(opts, :target_id)
    content_type = Keyword.get(opts, :content_type, "text")

    case AIModeration.check(content, content_type: content_type) do
      {:ok, %{action: :block, confidence: confidence, categories: categories}}
          when confidence >= @high_confidence_threshold ->
        primary_category = List.first(categories)

        if primary_category in @auto_action_categories do
          auto_enforce(user_id, target_type, target_id, primary_category, confidence)
        else
          flag_for_review(user_id, target_type, target_id, primary_category, confidence)
        end

      {:ok, %{action: :block, confidence: confidence, categories: categories}} ->
        flag_for_review(user_id, target_type, target_id, List.first(categories), confidence)

      {:ok, %{action: :flag, confidence: confidence, categories: categories}} ->
        flag_for_review(user_id, target_type, target_id, List.first(categories), confidence)

      {:ok, %{action: :allow, confidence: confidence}} ->
        AuditLogs.log(%{
          target_type: target_type, target_id: target_id,
          action: "ai_allow", ai_confidence: confidence, ai_action: "allow"
        })
        :ok
    end
  end

  defp auto_enforce(user_id, target_type, target_id, category, confidence) do
    # 1. Create report automatically
    reporter = Repo.get!(CGraph.Accounts.User, auto_reporter_id())
    {:ok, report} = Reports.create_report(reporter, %{
      target_type: target_type,
      target_id: target_id,
      category: category,
      description: "Auto-flagged by AI (#{category}, confidence: #{confidence})"
    })

    # 2. Apply enforcement based on category severity
    action = enforcement_action(category)
    system_user = Repo.get!(CGraph.Accounts.User, system_user_id())
    Enforcement.review_report(system_user, report.id, %{
      action: action,
      notes: "Auto-moderation action"
    })

    # 3. Log to audit trail
    AuditLogs.log(%{
      target_type: target_type, target_id: target_id, report_id: report.id,
      action: "ai_block", ai_category: category, ai_confidence: confidence,
      ai_action: "block", auto_actioned: true
    })
  end

  defp flag_for_review(user_id, target_type, target_id, category, confidence) do
    reporter = Repo.get!(CGraph.Accounts.User, auto_reporter_id())
    {:ok, report} = Reports.create_report(reporter, %{
      target_type: target_type,
      target_id: target_id,
      category: category,
      description: "Flagged by AI for review (#{category}, confidence: #{confidence})"
    })

    AuditLogs.log(%{
      target_type: target_type, target_id: target_id, report_id: report.id,
      action: "ai_flag", ai_category: category, ai_confidence: confidence,
      ai_action: "flag", auto_actioned: false
    })
  end

  defp enforcement_action(:spam), do: "mute_1h"
  defp enforcement_action(:hate_speech), do: "warn"
  defp enforcement_action(:violence), do: "remove_content"
  defp enforcement_action(:sexual), do: "remove_content"
  defp enforcement_action(_), do: "flag"

  defp system_user_id, do: Application.get_env(:cgraph, :system_user_id)
end
```

```elixir
defmodule CGraph.Moderation.Stats do
  # ... existing functions ...

  def reports_by_category(days \\ 30) do
    cutoff = DateTime.add(DateTime.utc_now(), -days * 86400)
    Report
    |> where([r], r.inserted_at >= ^cutoff)
    |> group_by([r], r.category)
    |> select([r], {r.category, count(r.id)})
    |> Repo.all()
    |> Map.new()
  end

  def reports_trend(days \\ 30, interval \\ :day) do
    cutoff = DateTime.add(DateTime.utc_now(), -days * 86400)
    Report
    |> where([r], r.inserted_at >= ^cutoff)
    |> group_by([r], fragment("date_trunc(?, ?)", ^Atom.to_string(interval), r.inserted_at))
    |> select([r], {fragment("date_trunc(?, ?)", ^Atom.to_string(interval), r.inserted_at), count(r.id)})
    |> order_by([r], fragment("date_trunc(?, ?)", ^Atom.to_string(interval), r.inserted_at))
    |> Repo.all()
  end

  def resolution_rate do
    total = Repo.aggregate(Report, :count)
    resolved = Report |> where([r], r.status in ["resolved", "dismissed"]) |> Repo.aggregate(:count)
    if total > 0, do: Float.round(resolved / total * 100, 1), else: 0.0
  end

  def moderator_leaderboard(limit \\ 10) do
    ReviewAction
    |> where([ra], ra.inserted_at >= ^thirty_days_ago())
    |> group_by([ra], ra.reviewer_id)
    |> select([ra], %{
      reviewer_id: ra.reviewer_id,
      actions_count: count(ra.id),
      last_action: max(ra.inserted_at)
    })
    |> order_by([ra], desc: count(ra.id))
    |> limit(^limit)
    |> Repo.all()
    |> Enum.map(&load_reviewer/1)
  end

  def ai_auto_action_stats(days \\ 30) do
    cutoff = DateTime.add(DateTime.utc_now(), -days * 86400)
    AuditLog
    |> where([l], l.inserted_at >= ^cutoff and not is_nil(l.ai_action))
    |> group_by([l], [l.ai_action, l.auto_actioned])
    |> select([l], {l.ai_action, l.auto_actioned, count(l.id)})
    |> Repo.all()
  end

  def appeals_outcome_stats(days \\ 30) do
    cutoff = DateTime.add(DateTime.utc_now(), -days * 86400)
    Appeal
    |> where([a], a.inserted_at >= ^cutoff)
    |> group_by([a], a.status)
    |> select([a], {a.status, count(a.id)})
    |> Repo.all()
    |> Map.new()
  end

  def comprehensive_stats do
    %{
      reports_today: reports_reviewed_today(),
      avg_response_time: average_response_time(),
      active_restrictions: active_restriction_count(),
      reports_by_category: reports_by_category(),
      reports_trend: reports_trend(),
      resolution_rate: resolution_rate(),
      moderator_leaderboard: moderator_leaderboard(),
      ai_stats: ai_auto_action_stats(),
      appeals_stats: appeals_outcome_stats()
    }
  end
end
```

```elixir
def batch_review(report_ids, reviewer_id, params) do
  action = Map.fetch!(params, "action")

  Repo.transaction(fn ->
    results = Enum.map(report_ids, fn id ->
      case Enforcement.review_report(id, reviewer_id, params) do
        {:ok, result} -> {:ok, id, result}
        {:error, reason} -> {:error, id, reason}
      end
    end)

    {successes, failures} = Enum.split_with(results, &match?({:ok, _, _}, &1))
    %{succeeded: length(successes), failed: length(failures), details: results}
  end)
end
```

```elixir
  def perform(%Oban.Job{args: %{"appeal_id" => appeal_id, "outcome" => outcome}}) do
    appeal = Repo.get!(Appeal, appeal_id) |> Repo.preload([:user, :report])

    template = case outcome do
      "approved" -> :appeal_approved
      "denied" -> :appeal_denied
    end

    CGraph.Workers.Orchestrator.enqueue(%{
      type: :email,
      template: template,
      to: appeal.user.email,
      assigns: %{
        user_name: appeal.user.display_name,
        appeal_reason: appeal.reason,
        outcome: outcome,
        reviewer_notes: appeal.reviewer_notes,
        restriction_type: appeal.report.target_type
      }
    })
  end
end
```
